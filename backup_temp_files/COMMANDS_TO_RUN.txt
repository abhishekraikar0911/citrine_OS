# === MANUAL TEST COMMANDS ===
# Copy and paste these one at a time into the terminal

# 1. GET CURRENT TRANSACTION COUNT
docker exec csms-postgres psql -U citrine -d citrine -c "SELECT id, \"transactionId\", \"isActive\" FROM \"Transactions\" WHERE \"stationId\"='250822008C06' ORDER BY id DESC LIMIT 1;"

# 2. SEND REMOTE START TRANSACTION
curl -X POST http://localhost:8081/ocpp/1.6/evdriver/remoteStartTransaction \
  -H "Content-Type: application/json" \
  -d '{
    "identifier": ["250822008C06"],
    "request": {
      "connectorId": 1,
      "idTag": "TEST123"
    },
    "tenantId": 1
  }'

# 3. WAIT A FEW SECONDS, THEN CHECK DATABASE AGAIN
sleep 5
docker exec csms-postgres psql -U citrine -d citrine -c "SELECT id, \"transactionId\", \"isActive\" FROM \"Transactions\" WHERE \"stationId\"='250822008C06' ORDER BY id DESC LIMIT 1;"

# 4. ONCE YOU HAVE A TRANSACTION ID FROM STEP 3, SEND REMOTE STOP
# Replace 2 with the actual transaction ID from step 3
curl -X POST http://localhost:8081/ocpp/1.6/evdriver/remoteStopTransaction \
  -H "Content-Type: application/json" \
  -d '{
    "identifier": ["250822008C06"],
    "request": {
      "transactionId": 2
    },
    "tenantId": 1
  }'

# 5. WAIT AND CHECK FINAL STATE
sleep 5
docker exec csms-postgres psql -U citrine -d citrine -c "SELECT id, \"transactionId\", \"isActive\", \"startTime\", \"endTime\" FROM \"Transactions\" WHERE \"stationId\"='250822008C06' ORDER BY id DESC LIMIT 1;"

# 6. CHECK SERVER LOGS FOR OUR NEW HANDLER MESSAGES
docker logs csms-core --since 1m | grep -i "remote\|accepted by station"
